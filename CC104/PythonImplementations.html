Array (basic operations)

# array_basic.py
class SimpleArray:
    def __init__(self, data=None):
        self.data = list(data) if data else []
    def append(self, x):
        self.data.append(x)
    def insert(self, idx, x):
        self.data.insert(idx, x)
    def delete(self, idx):
        return self.data.pop(idx)
    def find(self, x):
        try: return self.data.index(x)
        except ValueError: return -1
    def __str__(self):
        return str(self.data)

# usage
# a = SimpleArray([1,2,3])
# a.append(4); a.insert(1,9); print(a)

Stack + infix → postfix & evaluation

# stack_infix_postfix.py
precedence = {'+':1, '-':1, '*':2, '/':2, '^':3}

def infix_to_postfix(expr):
    stack = []
    out = []
    tokens = expr.split()
    for t in tokens:
        if t.isnumeric():
            out.append(t)
        elif t == '(':
            stack.append(t)
        elif t == ')':
            while stack and stack[-1] != '(':
                out.append(stack.pop())
            stack.pop()
        else:
            while stack and stack[-1] != '(' and precedence.get(stack[-1],0) >= precedence.get(t,0):
                out.append(stack.pop())
            stack.append(t)
    while stack:
        out.append(stack.pop())
    return ' '.join(out)

def eval_postfix(expr):
    stack = []
    for t in expr.split():
        if t.isnumeric():
            stack.append(int(t))
        else:
            b = stack.pop(); a = stack.pop()
            if t == '+': stack.append(a+b)
            elif t == '-': stack.append(a-b)
            elif t == '*': stack.append(a*b)
            elif t == '/': stack.append(a//b)
            elif t == '^': stack.append(a**b)
    return stack[0]

# usage:
# p = infix_to_postfix('( 3 + 4 ) * 5')
# print(p)  # -> 3 4 + 5 *
# print(eval_postfix(p)) # -> 35

Queue (simple)

# queue_simple.py
from collections import deque
class SimpleQueue:
    def __init__(self): self.q = deque()
    def enqueue(self, x): self.q.append(x)
    def dequeue(self): return self.q.popleft()
    def is_empty(self): return len(self.q)==0
    def peek(self): return self.q[0]

# usage: q = SimpleQueue(); q.enqueue(1); q.enqueue(2); print(q.dequeue())

Binary Tree + traversals

# binary_tree.py
class Node:
    def __init__(self, val):
        self.val = val
        self.left = None
        self.right = None

def inorder(root, res):
    if root:
        inorder(root.left, res)
        res.append(root.val)
        inorder(root.right, res)

def preorder(root, res):
    if root:
        res.append(root.val)
        preorder(root.left, res)
        preorder(root.right, res)

def postorder(root, res):
    if root:
        postorder(root.left, res)
        postorder(root.right, res)
        res.append(root.val)

# usage:
# r = Node(1); r.left=Node(2); r.right=Node(3)
# out=[]; inorder(r,out); print(out)
