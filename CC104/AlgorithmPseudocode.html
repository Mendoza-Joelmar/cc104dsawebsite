10 Pseudocode (one each)

1. Linear Search

FUNCTION linearSearch(A, key):
  FOR i FROM 0 TO length(A)-1:
    IF A[i] == key: RETURN i
  RETURN -1

2. Binary Search (sorted A)

FUNCTION binarySearch(A, key):
  low = 0; high = length(A)-1
  WHILE low <= high:
    mid = (low + high) // 2
    IF A[mid] == key: RETURN mid
    ELSE IF A[mid] < key: low = mid + 1
    ELSE: high = mid - 1
  RETURN -1

3. Bubble Sort

FUNCTION bubbleSort(A):
  FOR i FROM 0 TO n-2:
    FOR j FROM 0 TO n-2-i:
      IF A[j] > A[j+1]: SWAP(A[j], A[j+1])

4. Selection Sort

FUNCTION selectionSort(A):
  FOR i FROM 0 TO n-2:
    minIdx = i
    FOR j FROM i+1 TO n-1:
      IF A[j] < A[minIdx]: minIdx = j
    SWAP(A[i], A[minIdx])

5. Insertion Sort

FUNCTION insertionSort(A):
  FOR i FROM 1 TO n-1:
    key = A[i]
    j = i - 1
    WHILE j >= 0 AND A[j] > key:
      A[j+1] = A[j]
      j = j - 1
    A[j+1] = key

6. Merge Sort

FUNCTION mergeSort(A):
  IF length(A) <= 1: RETURN A
  mid = length(A)//2
  left = mergeSort(A[:mid])
  right = mergeSort(A[mid:])
  RETURN merge(left, right)

7. Quick Sort

FUNCTION quickSort(A, low, high):
  IF low < high:
    p = partition(A, low, high)
    quickSort(A, low, p-1)
    quickSort(A, p+1, high)

8. Dijkstra (sketch)

FUNCTION dijkstra(graph, src):
  dist[] = INFINITY; dist[src]=0
  PQ = priority queue with (src,0)
  WHILE PQ not empty:
    u = extract-min(PQ)
    FOR each neighbor v of u:
      alt = dist[u] + weight(u,v)
      IF alt < dist[v]: dist[v]=alt; update PQ
  RETURN dist

9. BFS

FUNCTION BFS(start):
  create queue Q
  mark start visited; enqueue(Q,start)
  WHILE Q not empty:
    u = dequeue(Q)
    FOR each neighbor v of u:
      IF not visited[v]: mark visited; enqueue(Q,v)

10. DFS (recursive)

FUNCTION DFS(u):
  mark u visited
  FOR each neighbor v of u:
    IF not visited[v]: DFS(v)
